# Settings file for numsim program
# Run ./numsim lid_driven_cavity.txt

# Problem description
physicalSizeX = 5.0   # physical size of the domain
physicalSizeY = 1.0
endTime = 20.0        # duration of the simulation
re = 1500             # Reynolds number
gX = 0.0              # external forces, set to (gX,gY) = (0,-9.81) to account for gravity
gY = 0.0


# Discretization parameters
nCellsX = 100        # number of cells in x and y direction
nCellsY = 20
useDonorCell = true   # if donor cell discretization should be used, possible values: true false
alpha = 0.5           # factor for donor-cell scheme, 0 is equivalent to central differences
tau = 0.5             # safety factor for time step width
maximumDt = 0.1       # maximum values for time step width

# Solver parameters
pressureSolver = SOR  # which pressure solver to use, possible values: GaussSeidel SOR CG
omega = 1.6           # overrelaxation factor, only for SOR solver
epsilon = 1e-5        # tolerance for 2-norm of residual
maximumNumberOfIterations = 1e4    # maximum number of iterations in the solver


# Dirichlet boundary conditions
# first entry: Velocity; oder Pressure;
# second entry: Dirichlet;
# third and fourth entry: x1; x2; (top, bottom) or y1; y2; (left, right) as area where boundary value is set
# fifth (and sixth) entry: u; v (Velocity) or p (Pressure)

bottomBound= Velocity; Dirichlet; 0.0; 5.0; 0.0; 0.0
topBound= Velocity; Dirichlet; 0.0; 5.0; 0.0; 0.0
leftBound= Velocity; Dirichlet; 0.5; 1.0; 1.0; 0.0
rightBound= Pressure; Dirichlet; 0.0; 1.0; 0.0; 0.0


# Objects
# first entry: Rectangle, Ellipse, Triangle
# for Rectangle: x(lower left corner); y(lower left corner); x(upper right corner); y(upper right corner)
# for Ellipse: x(center); y(center); a; b
# for Triangle: x(point 1); y(point 1); x(point 2); y(point 2); x(point 3); y(point 3)

object= Rectangle; 0.0; 0.0; 0.5; 0.5;
